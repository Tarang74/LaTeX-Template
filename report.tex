\section{Question 1}
Given
\begin{equation}
    f\left( x \right) =
    \begin{cases}
        x^3 & 0 \leqslant x < 2 \\
        -x  & 2 \leqslant x < 4 \\
        0   & \text{otherwise}.
    \end{cases}
\end{equation}
\begin{enumerate}[label=(\alph*)]
    \item Find the slope of \(f\left( x \right)\) at \(x = 1\).
          \begin{solution}
              The first derivative of \(f\left( x \right)\) is given by
              \begin{equation*}
                  f'\left( x \right) =
                  \begin{cases}
                      0                    & x \leqslant 0     \\
                      3x^2                 & 0 < x < 2         \\
                      -1                   & 2 < x < 4         \\
                      0                    & x > 4             \\
                      \text{indeterminate} & \text{otherwise}.
                  \end{cases}
              \end{equation*}
              At \(x = 1\), the slope of the function \(f\left( x \right)\) is
              \(3 \left( 1 \right)^2 = 3\).
          \end{solution}
\end{enumerate}
